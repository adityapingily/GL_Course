jobs = pd.DataFrame()
for job in parent_tags:
    title_tag = job.find_element_by_css_selector('span.job-title-text')
    title = title_tag.get_attribute('innerHTML').replace('<b>','').replace('</b>','')
    
    com_tag = job.find_element_by_css_selector('span.company-name-text')
    company = com_tag.get_attribute('innerHTML')
    
    loc_tag = job.find_element_by_css_selector('span.job-location')
    loc_tag = loc_tag.find_element_by_css_selector('span')
    location = loc_tag.get_attribute('innerHTML')
    
    des_tag = job.find_element_by_css_selector('div.job-description')
    description = des_tag.get_attribute('innerHTML')
    description = bs4.BeautifulSoup(description, 'html.parser').text
    #rep_str = '...<span class="job-attribution"> </span>'
    #description = description.replace(rep_str, '')
    
    
    #print(title,'|', company, '|', location, '|', description)
    curr_job = {'title': title,
                'company': company,
                'location': location,
                'description': description}
    jobs = jobs.append(curr_job, ignore_index= True)
